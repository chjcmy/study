***미리 방화벽을 해놓는 걸 추천한다

```bash
vim open_k8s_ports.sh
```

```bash
#!/bin/bash

# 필요한 포트 목록 (필요에 따라 추가)
ports=(
  6443     # Kubernetes API 서버
  10250    # Kubelet
  2375     # Docker (또는 Containerd)
  30000-32767 # NodePort 서비스
  80       # Dashboard, Ingress Controller
  443      # Ingress Controller
  44134    # Helm Tiller
  2379     # etcd
  2380     # etcd
)

# 방화벽 설정 함수
function allow_port {
  port="$1"
  ufw allow $port/tcp
  ufw allow $port/udp
}

# 모든 포트에 대해 방화벽 허용
for port in "${ports[@]}"; do
  if [[ $port == *"-"* ]]; then
    # 포트 범위인 경우
    start=$(echo $port | cut -d'-' -f1)
    end=$(echo $port | cut -d'-' -f2)
    for ((i=$start; i<=$end; i++)); do
      allow_port $i
    done
  else
    allow_port $port
  fi
done

# 방화벽 활성화 (필요한 경우)
# ufw enable

# 방화벽 상태 확인
ufw status
```

