fun main() {  
    val dog: Animal = Dog()  
    val cat = Cat()  
  
    println("dog is Animal: ${dog is Animal}")  
    println("cat is Animal: ${cat is Animal}")  
  
    if (dog is Animal) {  
        println("dog is Animal")  
    }  
}  
  
class Cat : Animal(), Drawable {  
    override fun draw() {  
        println("draw")  
    }  
  
    override fun move() {  
        super.move()  
        println("고양이가 움직인다")  
    }  
  
}  
  
class Dog : Animal() {  
    override fun move() {  
        super.move()  
        println("강아지가 움직인다")  
    }  
}  
  
interface Drawable {  
    fun draw()  
}  
  
abstract class Animal {  
    open fun move(){  
        println("움직인다")  
    }  
}
